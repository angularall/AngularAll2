Example-14:Display amount in currency format(usage of |currency)

  • {{tab}} 

<html>
	<head>
		<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.6.1/angular.min.js"></script> <!-- Step-1 -->
		<script>
			angular.module('myApp', []); <!--Step-2 -->
			angular.module('myApp').controller('mobileController',function($scope){ <!-- Step-3 -->
				$scope.mobileData = [{'brand':'Apple','price':'800','model':'7s'},
				                     {'brand':'Samsung','price':'650','model':'SJ4'},
				                     {'brand':'Blackberry','price':'750','model':'Dtek'}];
				$scope.sortHeader='brand';
				$scope.search={};
			});
		</script>
	</head>
	<body ng-app="myApp"><!-- Step-4 -->
		<div  ng-controller="mobileController"> <!-- Step-5 -->
			<h4>Search in Name:<input type="text" ng-model="search.brand"/></h4>
			<h4>Search in price:<input type="text" ng-model="search.price"/></h4>
			<h4>Search in model:<input type="text" ng-model="search.model"/></h4>
			<h4>Search in ALL :<input type="text" ng-model="search.$"/></h4>
			<hr>
			<table>
      			<tr>
        			<th><a ng-click="sortHeader='brand'">Brand</a></th>
        			<th><a ng-click="sortHeader='price'">Price</a></th>
        			<th><a ng-click="sortHeader='model'">Model</a></th>
      			</tr>
      				<tr ng-repeat="(index,mobile) in mobileData|orderBy:sortHeader|filter:search"><!-- Step-6 -->
        				<td>{{mobile.brand}}</td>
        				<td>{{mobile.price|currency:'$'}}</td>
        				<td>{{mobile.model}}</td>
      				</tr>
    		</table>
    		<hr>
		</div>
	</body>
</html>

New concepts are below: goto All Steps for complete info
  • |currency filter is used for display amounts in currency format in default
  • |currency:symbol will provide the currency symbol.
