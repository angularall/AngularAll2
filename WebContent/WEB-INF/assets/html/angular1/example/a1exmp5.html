Example-5: Print list of data by using angular looping with ng-repeat directive(usage of ng-repeat in looping)

  • {{tab}} 

<html>
	<head>
		<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.6.1/angular.min.js"></script> <!-- Step-1 -->
		<script>
			//define module
			angular.module('myApp', []); <!--Step-2 -->
			//define controller <!-- Step-3 -->
			angular.module('myApp').controller('mobileController',function($scope){
				$scope.mobileBrandList = ['Apple', 'Samsung', 'Nexes','BlackBerry','HTC'];
			});
		</script>
	</head>
	<body ng-app="myApp"><!-- Step-4 -->
		<div  ng-controller="mobileController"> <!-- Step-5 -->
			<ul>
				<li ng-repeat="mobileBrand in mobileBrandList"><!-- Step-6 -->
					{ {mobileBrand}}
				</li>
			</ul>
		</div>
	</body>
</html>
